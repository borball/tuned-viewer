# Debug commands to run in the tuned-viewer pod

# 1. Check environment
oc exec -it deployment/tuned-viewer -n tuned-viewer -- python3 -m tuned_viewer env

# 2. Check what directories exist
oc exec -it deployment/tuned-viewer -n tuned-viewer -- ls -la /etc/tuned/ 2>/dev/null || echo "/etc/tuned not found"
oc exec -it deployment/tuned-viewer -n tuned-viewer -- ls -la /usr/lib/tuned/ 2>/dev/null || echo "/usr/lib/tuned not found"
oc exec -it deployment/tuned-viewer -n tuned-viewer -- ls -la /host/etc/tuned/ 2>/dev/null || echo "/host/etc/tuned not found"
oc exec -it deployment/tuned-viewer -n tuned-viewer -- ls -la /host/usr/lib/tuned/ 2>/dev/null || echo "/host/usr/lib/tuned not found"

# 3. Check cluster status to see tuned pods
oc exec -it deployment/tuned-viewer -n tuned-viewer -- python3 -m tuned_viewer cluster

# 4. Try syncing profiles from cluster first
oc exec -it deployment/tuned-viewer -n tuned-viewer -- python3 -m tuned_viewer sync

# 5. Then list profiles again
oc exec -it deployment/tuned-viewer -n tuned-viewer -- python3 -m tuned_viewer list

# 6. Get node names for node analysis
oc get nodes --no-headers -o custom-columns=NAME:.metadata.name
